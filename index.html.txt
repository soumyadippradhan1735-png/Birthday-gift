  <!-- ================= LOGIC & INTERACTIVE SCRIPT ================= -->
  <script>
    // ================= 1. SYSTEM INITIALIZATION =================
    let currentStepNum = 1;
    const totalSteps = 5;

    // Trigger initial welcome card animations on load
    window.addEventListener('DOMContentLoaded', () => {
      gsap.fromTo("#step-1", 
        { opacity: 0, scale: 0.94, y: 30 }, 
        { opacity: 1, scale: 1, y: 0, duration: 1, ease: "power3.out" }
      );
      gsap.fromTo("#step-1 .animate-item", 
        { opacity: 0, y: 20 }, 
        { opacity: 1, y: 0, duration: 0.8, stagger: 0.12, ease: "power2.out", delay: 0.3 }
      );
    });

    // ================= 2. CARD STEP TRANSITION LOGIC =================
    function goToStep(nextNum) {
      if (nextNum > totalSteps) return;

      const currentEl = document.querySelector(`#step-${currentStepNum}`);
      const nextEl = document.querySelector(`#step-${nextNum}`);

      // Update Top Progress Bar
      const progressPercent = (nextNum / totalSteps) * 100;
      document.getElementById('progress-bar').style.width = `${progressPercent}%`;

      // Animate Active Card Out
      gsap.to(currentEl, {
        opacity: 0,
        scale: 0.94,
        y: -20,
        duration: 0.45,
        ease: "power2.inOut",
        onComplete: () => {
          currentEl.classList.add('hidden');
          nextEl.classList.remove('hidden');

          // Animate New Card In
          gsap.fromTo(nextEl,
            { opacity: 0, scale: 0.96, y: 20 },
            { opacity: 1, scale: 1, y: 0, duration: 0.55, ease: "power2.out" }
          );

          // Stagger Entrance of Inner Elements
          const children = nextEl.querySelectorAll('.animate-item');
          if (children.length > 0) {
            gsap.fromTo(children,
              { opacity: 0, y: 15 },
              { opacity: 1, y: 0, duration: 0.5, stagger: 0.1, ease: "power2.out", delay: 0.1 }
            );
          }

          currentStepNum = nextNum;
        }
      });
    }

    // ================= 3. POLAROID PARALLAX EFFECT =================
    const polaroid = document.getElementById('polaroid');
    if (polaroid) {
      polaroid.addEventListener('mousemove', (e) => {
        const rect = polaroid.getBoundingClientRect();
        const x = e.clientX - rect.left - rect.width / 2;
        const y = e.clientY - rect.top - rect.height / 2;
        
        // Tilt the polaroid matching cursor location
        gsap.to(polaroid, {
          rotateY: x * 0.12,
          rotateX: -y * 0.12,
          transformPerspective: 800,
          duration: 0.3,
          ease: "power2.out"
        });
      });

      polaroid.addEventListener('mouseleave', () => {
        // Return back to default tilt
        gsap.to(polaroid, {
          rotateY: -2,
          rotateX: 0,
          duration: 0.6,
          ease: "power2.out"
        });
      });
    }

    // ================= 4. THREE.JS 3D BACKGROUND ENGINE =================
    const canvas = document.getElementById('three-canvas');
    const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 100);
    camera.position.z = 20;

    // Mathematical definition of a symmetric Heart Shape
    const heartShape = new THREE.Shape();
    heartShape.moveTo(0, 0.4);
    heartShape.bezierCurveTo(0, 0.4, 0.35, 1, 0.9, 1);
    heartShape.bezierCurveTo(1.5, 1, 1.8, 0.4, 1.8, 0);
    heartShape.bezierCurveTo(1.8, -0.7, 1.1, -1.3, 0, -2);
    heartShape.bezierCurveTo(-1.1, -1.3, -1.8, -0.7, -1.8, 0);
    heartShape.bezierCurveTo(-1.8, 0.4, -1.5, 1, -0.9, 1);
    heartShape.bezierCurveTo(-0.35, 1, 0, 0.4, 0, 0.4);

    const extrudeSettings = {
      depth: 0.3,
      bevelEnabled: true,
      bevelSegments: 4,
      steps: 1,
      bevelSize: 0.08,
      bevelThickness: 0.08
    };
    const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
    geometry.center();

    // Spawn 3D Heart Array
    const hearts = [];
    const numHearts = 22;

    for (let i = 0; i < numHearts; i++) {
      // Modern Glass/Gloss physical material parameters
      const material = new THREE.MeshPhysicalMaterial({
        color: new THREE.Color().setHSL(Math.random() * 0.06 + 0.94, 0.85, 0.55), // Vibrant red-pink spectrum
        roughness: 0.15,
        metalness: 0.1,
        clearcoat: 1.0,
        clearcoatRoughness: 0.1,
        transparent: true,
        opacity: 0.75,
        depthWrite: false // Mitigates depth sorting anomalies of transparent layers
      });

      const mesh = new THREE.Mesh(geometry, material);

      // Random dispersion within 3D Space
      const x = (Math.random() - 0.5) * 24;
      const y = (Math.random() - 0.5) * 16;
      const z = (Math.random() - 0.5) * 8 - 1;

      mesh.position.set(x, y, z);

      const scale = Math.random() * 0.4 + 0.35;
      mesh.scale.set(scale, scale, scale);
      mesh.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, Math.random() * Math.PI);

      scene.add(mesh);

      hearts.push({
        mesh,
        material,
        basePos: new THREE.Vector3(x, y, z),
        speed: Math.random() * 0.8 + 0.4,
        phase: Math.random() * Math.PI * 2,
        rotSpeed: new THREE.Vector3(
          (Math.random() - 0.5) * 0.01,
          (Math.random() - 0.5) * 0.01,
          (Math.random() - 0.5) * 0.01
        ),
        amplitude: Math.random() * 0.5 + 0.25
      });
    }

    // Dynamic 3D lighting coordinates
    const ambientLight = new THREE.AmbientLight(0xffffff, 0.45);
    scene.add(ambientLight);

    const dirLight = new THREE.DirectionalLight(0xffb6c1, 1.2);
    dirLight.position.set(5, 10, 7);
    scene.add(dirLight);

    const pointLight = new THREE.PointLight(0xff007f, 1.5, 30);
    pointLight.position.set(-5, -5, 5);
    scene.add(pointLight);

    let heartsSwirlingOut = false;
    let clock = new THREE.Clock();

    function animateThree() {
      requestAnimationFrame(animateThree);
      const time = clock.getElapsedTime();

      hearts.forEach(heart => {
        if (!heartsSwirlingOut) {
          // Floating wave animations
          heart.mesh.position.y = heart.basePos.y + Math.sin(time * heart.speed + heart.phase) * heart.amplitude;
          heart.mesh.position.x = heart.basePos.x + Math.cos(time * (heart.speed * 0.5) + heart.phase) * (heart.amplitude * 0.3);

          heart.mesh.rotation.x += heart.rotSpeed.x;
          heart.mesh.rotation.y += heart.rotSpeed.y;
          heart.mesh.rotation.z += heart.rotSpeed.z;
        } else {
          // Grand Finale Swirl Sequence
          const dir = new THREE.Vector3(heart.mesh.position.x, 0, heart.mesh.position.z).normalize();
          heart.mesh.position.addScaledVector(dir, 0.22); // Drift outwards
          heart.mesh.position.y += 0.28; // Ascend off camera

          heart.mesh.rotation.x += 0.06;
          heart.mesh.rotation.y += 0.06;

          heart.material.opacity -= 0.012; // Gradually vanish
          if (heart.material.opacity < 0) {
            heart.material.opacity = 0;
          }
        }
      });

      renderer.render(scene, camera);
    }

    animateThree();

    // Canvas Resize Observer
    window.addEventListener('resize', () => {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });

    // ================= 5. GRAND FINALE SEQUENCING =================
    const celebrateBtn = document.getElementById('celebrate-btn');
    if (celebrateBtn) {
      celebrateBtn.addEventListener('click', () => {
        // Step 5.1: Disable and fade the trigger button
        gsap.to(celebrateBtn, {
          opacity: 0,
          scale: 0.9,
          pointerEvents: 'none',
          duration: 0.4,
          onComplete: () => {
            celebrateBtn.style.display = 'none';
          }
        });

        // Step 5.2: Smoothly bring the final romantic text into viewport
        const finalMessage = document.getElementById('final-message');
        finalMessage.classList.remove('hidden');
        gsap.fromTo(finalMessage,
          { opacity: 0, y: 30, scale: 0.95 },
          { opacity: 1, y: 0, scale: 1, duration: 1.2, ease: "elastic.out(1, 0.75)", delay: 0.4 }
        );

        // Step 5.3: Trigger WebGL background swirl sequence
        heartsSwirlingOut = true;

        // Step 5.4: Initiate 5-Second Confetti Showers
        // Initial powerful corner bursts
        confetti({
          particleCount: 140,
          spread: 80,
          origin: { x: 0, y: 0.85 },
          colors: ['#f43f5e', '#ec4899', '#f472b6', '#fbbf24']
        });
        confetti({
          particleCount: 140,
          spread: 80,
          origin: { x: 1, y: 0.85 },
          colors: ['#f43f5e', '#ec4899', '#f472b6', '#fbbf24']
        });

        // Continuous shower generator loop
        const durationLimit = Date.now() + 5000;
        const showerInterval = setInterval(() => {
          if (Date.now() > durationLimit) {
            return clearInterval(showerInterval);
          }

          // Ambient side showers
          confetti({
            particleCount: 20,
            angle: 60,
            spread: 55,
            origin: { x: 0, y: 0.85 },
            colors: ['#f43f5e', '#ec4899', '#3b82f6']
          });
          confetti({
            particleCount: 20,
            angle: 120,
            spread: 55,
            origin: { x: 1, y: 0.85 },
            colors: ['#f43f5e', '#ec4899', '#3b82f6']
          });

          // Floating Custom Emoji Confetti
          try {
            confetti({
              particleCount: 6,
              scalar: 2.5,
              shapes: [
